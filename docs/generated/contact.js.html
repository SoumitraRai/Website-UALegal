/**
 * Contact Page JavaScript for UA Legal Website
 * This file contains functionality specific to the contact page.
 * 
 * @file contact.js
 * @author UA Legal
 * @version 1.0.0
 */

document.addEventListener('DOMContentLoaded', function() {
    // Initialize contact form
    initContactForm();
    
    // Initialize FAQ accordion
    initFaqAccordion();
});

/**
 * Initializes the contact form functionality
 */
function initContactForm() {
    // Get form element
    const contactForm = document.getElementById('contact-form');
    
    // Get message elements
    const formMessage = document.getElementById('form-message');
    const closeMessageBtn = document.getElementById('close-message');
    
    // Exit if any required element doesn't exist
    if (!contactForm || !formMessage || !closeMessageBtn) return;
    
    // Add submit event to form
    contactForm.addEventListener('submit', function(e) {
        // Prevent actual form submission
        e.preventDefault();
        
        // Get form inputs
        const name = contactForm.querySelector('#name').value.trim();
        const email = contactForm.querySelector('#email').value.trim();
        const phone = contactForm.querySelector('#phone').value.trim();
        const subject = contactForm.querySelector('#subject').value.trim();
        const practiceArea = contactForm.querySelector('#practice-area').value;
        const message = contactForm.querySelector('#message').value.trim();
        
        // Simple validation
        if (!name || !email || !phone || !subject || !practiceArea || !message) {
            alert('Please fill in all required fields.');
            return;
        }
        
        // Simple email validation
        const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if (!emailPattern.test(email)) {
            alert('Please enter a valid email address.');
            return;
        }
        
        // In a real-world scenario, you would send this data to a server
        // Here, we'll simulate a successful submission
        
        // Show success message
        formMessage.classList.add('active');
        
        // Reset form
        contactForm.reset();
    });
    
    // Close message when clicking close button
    closeMessageBtn.addEventListener('click', function() {
        formMessage.classList.remove('active');
    });
    
    // Close message when clicking outside
    window.addEventListener('click', function(e) {
        if (e.target === formMessage) {
            formMessage.classList.remove('active');
        }
    });
    
    // Close message with escape key
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape' && formMessage.classList.contains('active')) {
            formMessage.classList.remove('active');
        }
    });
}

/**
 * Initializes the FAQ accordion functionality
 */
function initFaqAccordion() {
    // Get all FAQ questions
    const faqQuestions = document.querySelectorAll('.faq-question');
    
    // Exit if no FAQ questions
    if (!faqQuestions.length) return;
    
    // Add click event to each FAQ question
    faqQuestions.forEach(question => {
        question.addEventListener('click', function() {
            // Get target answer ID
            const targetId = this.getAttribute('data-target');
            const answer = document.getElementById(targetId);
            
            // Exit if answer not found
            if (!answer) return;
            
            // Toggle active class on question
            this.classList.toggle('active');
            
            // Toggle height of answer
            if (answer.style.maxHeight) {
                answer.style.maxHeight = null;
                // Change + to -
                this.querySelector('.question-icon i').classList.remove('fa-minus');
                this.querySelector('.question-icon i').classList.add('fa-plus');
            } else {
                answer.style.maxHeight = answer.scrollHeight + 'px';
                // Change - to +
                this.querySelector('.question-icon i').classList.remove('fa-plus');
                this.querySelector('.question-icon i').classList.add('fa-minus');
            }
        });
    });
}
