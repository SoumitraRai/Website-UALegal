/**
 * Home Page JavaScript for UA Legal Website
 * This file contains functionality specific to the home page.
 * 
 * @file home.js
 * @author Your Name
 * @version 1.0.0
 */

/**
 * Executes when the DOM is fully loaded
 */
document.addEventListener('DOMContentLoaded', function() {
    // Initialize testimonials slider
    initTestimonialSlider();
});

/**
 * Testimonial data for the slider
 * @type {Array<Object>}
 */
const testimonials = [
    {
        quote: "UA Legal provided exceptional counsel during our company's merger. Their attention to detail and strategic approach was invaluable.",
        name: "Rajesh Mehta",
        position: "CEO, Mehta Enterprises"
    },
    {
        quote: "The team's thorough understanding of India's evolving legal landscape helped us navigate complex regulatory challenges with confidence.",
        name: "Priya Singh",
        position: "Managing Director, Singh Industries"
    },
    {
        quote: "Their expertise in intellectual property law was crucial in protecting our innovations in a highly competitive market.",
        name: "Vikram Patel",
        position: "Founder, TechInnovate"
    }
];

/**
 * Current index of displayed testimonial
 * @type {number}
 */
let currentTestimonialIndex = 0;

/**
 * Initializes the testimonial slider functionality
 */
function initTestimonialSlider() {
    // Get slider elements
    const testimonialSlider = document.getElementById('testimonial-slider');
    const prevButton = document.getElementById('prev-testimonial');
    const nextButton = document.getElementById('next-testimonial');
    const indicatorsContainer = document.getElementById('testimonial-indicators');
    
    // Exit if any element doesn't exist
    if (!testimonialSlider || !prevButton || !nextButton || !indicatorsContainer) return;
    
    // Create indicator dots
    testimonials.forEach((_, index) => {
        const indicator = document.createElement('div');
        indicator.className = 'testimonial-indicator' + (index === 0 ? ' active' : '');
        indicator.addEventListener('click', () => showTestimonial(index));
        indicatorsContainer.appendChild(indicator);
    });
    
    // Add click event for previous button
    prevButton.addEventListener('click', showPreviousTestimonial);
    
    // Add click event for next button
    nextButton.addEventListener('click', showNextTestimonial);
    
    // Initial render
    renderCurrentTestimonial();
    
    // Auto rotate testimonials every 7 seconds
    setInterval(showNextTestimonial, 7000);
}

/**
 * Shows the previous testimonial
 */
function showPreviousTestimonial() {
    currentTestimonialIndex = (currentTestimonialIndex - 1 + testimonials.length) % testimonials.length;
    renderCurrentTestimonial();
    updateIndicators();
}

/**
 * Shows the next testimonial
 */
function showNextTestimonial() {
    currentTestimonialIndex = (currentTestimonialIndex + 1) % testimonials.length;
    renderCurrentTestimonial();
    updateIndicators();
}

/**
 * Shows a specific testimonial by index
 * @param {number} index - The index of the testimonial to show
 */
function showTestimonial(index) {
    currentTestimonialIndex = index;
    renderCurrentTestimonial();
    updateIndicators();
}

/**
 * Updates the active indicator
 */
function updateIndicators() {
    const indicators = document.querySelectorAll('.testimonial-indicator');
    indicators.forEach((indicator, index) => {
        if (index === currentTestimonialIndex) {
            indicator.classList.add('active');
        } else {
            indicator.classList.remove('active');
        }
    });
}

/**
 * Renders the current testimonial in the slider
 */
function renderCurrentTestimonial() {
    const testimonialSlider = document.getElementById('testimonial-slider');
    if (!testimonialSlider) return;
    
    const current = testimonials[currentTestimonialIndex];
    
    // Create new testimonial with fade effect
    const newTestimonial = document.createElement('div');
    newTestimonial.className = 'testimonial';
    newTestimonial.style.opacity = '0';
    
    newTestimonial.innerHTML = `
        <div class="quote">"${current.quote}"</div>
        <div class="client">
            <div class="client-name">${current.name}</div>
            <div class="client-position">${current.position}</div>
        </div>
    `;
    
    // Replace old testimonial with fade effect
    testimonialSlider.innerHTML = '';
    testimonialSlider.appendChild(newTestimonial);
    
    // Trigger fade in effect
    setTimeout(() => {
        newTestimonial.style.transition = 'opacity 0.5s ease';
        newTestimonial.style.opacity = '1';
    }, 10);
}
